    // Função para redirecionar para um link aleatório
    function redirecionarAleatorio() {
        const linkAleatorio = destinos[Math.floor(Math.random() * destinos.length)];
        window.location.href = linkAleatorio;
    }

    // Exibir animação do círculo quando a página for carregada
    window.addEventListener('scroll', function() {
        const alturaPagina = document.documentElement.scrollHeight;
        const posicaoTela = window.innerHeight + window.scrollY;
        
        // Quando atingir o final da página, redirecionar e mostrar a animação
        if (posicaoTela >= alturaPagina) {
            document.getElementById("roseAnimation").style.display = "block"; // Exibe a animação
            setTimeout(redirecionarAleatorio, 20); // Redireciona após 2 segundos
        }
    });

    // Gerar números aleatórios
    function gerarNumerosAleatorios() {
        let numeros = "";
        for (let i = 0; i < 10; i++) {
            numeros += Math.floor(Math.random() * 100) + " ";
        }
        document.getElementById("randomNumbers").textContent = numeros;
    }

    // Gerar letras aleatórias
    function gerarLetrasAleatorias() {
        const letras = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz";
        let letrasAleatorias = "";
        for (let i = 0; i < 10; i++) {
            letrasAleatorias += letras.charAt(Math.floor(Math.random() * letras.length)) + " ";
        }
        document.getElementById("randomLetters").textContent = letrasAleatorias;
    }

    // Chamar funções para gerar números e letras aleatórias
    gerarNumerosAleatorios();
    gerarLetrasAleatorias();
